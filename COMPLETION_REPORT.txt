🚀 INSTALLATION & USAGE
══════════════════════════════════════════════════════════════════════════════

BUILD PACKAGE
  bash build.sh
  → Creates: dist/easypanel_1.0.0_all.deb

INSTALL PACKAGE
  sudo dpkg -i dist/easypanel_1.0.0_all.deb

RUN INSTALLATION WIZARD
  sudo easypanel install

INTERACTIVE MENU
  sudo easypanel

DIRECT COMMANDS
  sudo easypanel domains      # Domain management
  sudo easypanel dns         # DNS management
  sudo easypanel mail        # Mail management
  sudo easypanel databases   # Database management
  sudo easypanel cron        # Cron jobs
  sudo easypanel backup      # Backup & restore
  sudo easypanel status      # System status
  sudo easypanel help        # Help
  sudo easypanel version     # Version
